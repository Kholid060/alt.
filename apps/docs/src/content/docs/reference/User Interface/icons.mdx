import { UiIcons } from '@altdot/ui';

List of built-in icons that you can use.

<div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-8">
  {Object.entries(UiIcons).map(([name, Icon]) => (
    <div key={name} className="flex flex-col items-center justify-center !m-0 text-muted-foreground hover:text-foreground transition-colors">
      <Icon className="size-8" />
      <p className="text-sm text-nowrap !mt-1">{name}</p>
    </div>
  ))}
</div>

**Props**

| Name | Type | Description |
| ----------- | ----------- | ----------- |
| `size` | `?number` | The icon height and width size. Default to `24` |
| `strokeWidth` | `?number` | The icon's stroke width. Default to `2` |
| `color` | `?string` | The icon's color. Default to `currentColor` |
